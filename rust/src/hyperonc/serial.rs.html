42/// @struct c_to_rust_serializer_t

45/// @see gnd_api_t

62/// @struct serial_result_t
63/// @brief The result of a `serialize` operation reported by serializer
64/// @ingroup serializer_group
65/// @see gnd_api_t
66///
67#[repr(C)]
68pub enum serial_result_t {
69    /// @brief Successful serialization
70    ///
71    OK,
72    /// @brief Serialization of the value is not supported by serializer
73    ///
74    NOT_SUPPORTED,
75}
76
77impl From<serial::Result> for serial_result_t {
78    fn from(result: serial::Result) -> Self {
79        match result {
80            Ok(()) => serial_result_t::OK,
81            Err(serial::Error::NotSupported) => serial_result_t::NOT_SUPPORTED,
82        }
83    }
84}
85
86impl From<serial_result_t> for serial::Result {
87    fn from(result: serial_result_t) -> Self {
88        match result {
89            serial_result_t::OK => Ok(()),
90            serial_result_t::NOT_SUPPORTED => Err(serial::Error::NotSupported),
91        }
92    }
93}

102#[no_mangle]
103extern "C" fn serialize_bool_rust_adapter(context: *mut c_void, v: bool) -> serial_result_t {
104    let target = unsafe{ &mut*(context as *mut c_to_rust_serializer_t) }.borrow_mut();
105    target.serialize_bool(v).into()

126/// @struct RustToCSerializer
127/// @brief Adapt a serializer implemented in C to Rust API
128/// @ingroup serializer_group
129/// @see gnd_api_t
130///
131pub(crate) struct RustToCSerializer {
132    api: *const serializer_api_t,
133    context: *mut c_void,
134}
135
136impl serial::Serializer for RustToCSerializer {
137    fn serialize_bool(&mut self, v: bool) -> serial::Result {
138        self.call_serialize(self.api().serialize_bool, v)
139    }
140    fn serialize_i64(&mut self, v: i64) -> serial::Result {
141        self.call_serialize(self.api().serialize_longlong, v)
142    }
143    fn serialize_f64(&mut self, v: f64) -> serial::Result {
144        self.call_serialize(self.api().serialize_double, v)
145    }
146    fn serialize_str(&mut self, v: &str) -> serial::Result {
147        self.call_serialize(self.api().serialize_str, str_as_cstr(v).as_ptr())
148    }
149}
150
151impl RustToCSerializer {
152    pub fn new(api: *const serializer_api_t, context: *mut c_void) -> Self {
153        Self{ api, context }
154    }
155    fn api(&self) -> &serializer_api_t {
156        unsafe{ &*self.api }
157    }
158    fn call_serialize<T>(&self, serialize: Option<serial_serialize_func_t<T>>, v: T) -> serial::Result {
159        serialize.map_or(Err(serial::Error::NotSupported), |serialize| {
160            serialize(self.context, v).into()
161        })
162    }
