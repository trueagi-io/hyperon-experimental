24pub(crate) fn write_into_buf<T: std::fmt::Display>(obj: T, buf: *mut c_char, buf_len: usize) -> usize {
25
26    //If buf_len == 0, the caller is just interested in the size of buffer they will need
27    if buf_len == 0 {
28        struct LengthTracker(usize);
29        impl Write for LengthTracker {
30            fn write(&mut self, slice: &[u8]) -> Result<usize, std::io::Error> {
31                self.0 += slice.len();
32                Ok(slice.len())
33            }
34            fn flush(&mut self) -> Result<(), std::io::Error> { Ok (())}
35        }
36
37        let mut length_tracker = LengthTracker(0);
38        write!(length_tracker, "{obj}").unwrap();
39        length_tracker.0
40    } else {
41        //We are goint to try and actually render the object into the buffer, saving room for the terminator
42        let slice = unsafe{ slice::from_raw_parts_mut(buf as *mut u8, buf_len) };
43        let mut cursor = Cursor::new(slice);
44        let len = if let Err(_err) = write!(cursor, "{obj}") {
45            //The buffer was probably too short, so figure out the buffer size we need
46            cursor.into_inner()[0] = 0;
47            return write_into_buf(obj, buf, 0);
48        } else {
49            let len = cursor.position() as usize;
50            //We still need room for the terminator
51            if len == buf_len {
52                cursor.into_inner()[0] = 0;
53                return write_into_buf(obj, buf, 0)
54            } else {
55                len
56            }
57        };
58
59        //Write the terminator
60        cursor.into_inner()[len] = 0;
61        len
62    }
63}

65pub(crate) fn write_debug_into_buf<T: std::fmt::Debug>(obj: T, buf: *mut c_char, buf_len: usize) -> usize {
66    struct DisplayDebug<DebugT>(DebugT);
67    impl<DebugT: std::fmt::Debug> std::fmt::Display for DisplayDebug<DebugT> {
68        fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
69            write!(f, "{:?}", self.0)
70        }
71    }
72    write_into_buf(DisplayDebug(obj), buf, buf_len)
73}

105/// @struct write_t
106/// @brief A handle to a Rust std::fmt::Writer to be used from C code
107/// @ingroup misc_group
108///
109#[repr(C)]
110pub struct write_t(*const c_void);
111
112/// @brief Write C string into a Rust writer
113/// @ingroup misc_group
114/// @param[in]  cwrite  A handle to a Rust writer
115/// @param[in]  text  C string in UTF-8 format to be written
116/// @return 0 if string is written successfully, non-zero otherwise
117///
118#[no_mangle]
119pub extern "C" fn write_str(cwrite: write_t, text: *const c_char) -> isize {
120    let cwrite = unsafe{ &mut *(cwrite.0 as *mut CWrite) };
121    cwrite.res = cwrite.res.and_then(|()| cwrite.write.write_str(cstr_as_str(text)));
122    match cwrite.res {
123        Ok(()) => 0,
124        Err(_) => -1,
125    }
126}

128/// Rust wrapper of reference of the [std::fmt::Write] implementer which is used
129/// to writer to the C code and get the result back. Typical use-case is implementing
130/// [std::fmt::Display] or [std::fmt::Debug] for the C object.
131/// See [crate::metta::module_loader_t].
132pub struct CWrite<'a> {
133    /// The reference to the instance of the [std::fmt::Write] trait
134    write: &'a mut dyn std::fmt::Write,
135    /// The result of the last write operation invoked by C code
136    res: std::fmt::Result,
137}
138
139impl<'a> CWrite<'a> {
140    /// Create new writer from the reference to the [std::fmt::Write] implementation
141    pub fn new<T: std::fmt::Write>(write: &'a mut T) -> Self {
142        Self{ write, res: Ok(()) }
143    }
144    /// Get the result of the last operation performed on writer
145    pub fn res(&self) -> std::fmt::Result {
146        self.res
147    }
148    /// Call passed operator in a context of the writer passing instance of the
149    /// [CWrite] as an argument and returning the result back
150    pub fn with<F: 'a + FnOnce(&mut Self)>(&mut self, f: F) -> std::fmt::Result {
151        f(self);
152        self.res()
153    }
154}
155
156impl<'a>  From<&mut CWrite<'a>> for write_t {
157    fn from(write: &mut CWrite<'a>) -> Self {
158        Self((write as *mut CWrite).cast())
159    }
