41/// @brief A C representation of the Rust [ModuleLoader] interface. User can
42/// provide [ModuleLoader] methods implemented in C.
43/// @ingroup module_group
44///
45#[repr(C)] 
46pub struct module_loader_t {
47    /// @brief A function to load the module my making MeTTa API calls.
48    /// @param[in]  payload  The module loader self pointer
49    /// @param[in]  run_context  The `run_context_t` to provide access to the MeTTa run interface
50    /// @param[in]  err  The writer to fill with error message if any
51    /// @return 0 if success, non-zero otherwise; code should put the explanation text
52    /// to the `err` field.
53    load: Option<extern "C" fn(payload: *const c_void, context: *mut run_context_t, err: write_t) -> isize>,
54    /// @brief Loads module's tokens into target module. This method is used for both
55    /// initial token loading and exporting module's tokens into importing
56    /// module.
57    /// @param[in]  payload  The module loader self pointer
58    /// @param[in]  target  The module to load tokens into
59    /// @param[in]  metta  The context MeTTa runner
60    /// @param[in]  err  The writer to fill with error message if any
61    /// @return 0 if success, non-zero otherwise; code should put the explanation text
62    /// to the `err` field.
63    load_tokens: Option<extern "C" fn(payload: *const c_void, target: metta_mod_ref_t, metta: metta_t, err: write_t) -> isize>,
64    /// @brief Prints module loader content as a string, used for implementing
65    /// [std::fmt::Debug].
66    /// @param[in]  payload  The module loader self pointer
67    /// @param[in]  text  Object to write the text into
68    to_string: Option<extern "C" fn(payload: *const c_void, text: write_t)>,
69    /// @brief Frees module loader and all associated memory
70    /// @param[in]  payload  The module loader self pointer
71    free: Option<extern "C" fn(payload: *mut c_void)>,
72}

74/// The wrapper of the module_loader_t providing Rust API of the C implementation
75pub struct CModuleLoader {
76    payload: *const module_loader_t,
77}

79//FUTURE TODO.  See QUESTION around CFsModuleFormat about whether we trust the C plugins to be reentrant
80unsafe impl Send for CModuleLoader {}
81unsafe impl Sync for CModuleLoader {}
82
83impl CModuleLoader {
84    /// Create new Rust API for the C module loader object
85    pub fn new(cloader: *const module_loader_t) -> Self {
86        assert!(!cloader.is_null(), "ModuleLoader::load() implementation is required");
87        Self{ payload: cloader }
88    }
89
90    fn result(&self, rc: isize, err: String) -> Result<(), String> {
91        if rc == 0 {
92            Ok(())
93        } else {
94            if err.is_empty() {
95                Err("Unexpected error while loading module".into())
96            } else {
97                Err(format!("Error loading module: loader: {}, error: {}", self, err))
98            }
99        }
100    }
101
102    fn payload(&self) -> *const c_void {
103        self.payload.cast()
104    }
105
106    fn reference(&self) -> &module_loader_t {
107        unsafe{ &*self.payload }
108    }
109}
110
111impl Drop for CModuleLoader {
112    fn drop(&mut self) {
113        match self.reference().free {
114            Some(free) => free(self.payload().cast_mut()),
115            None => {},
116        }
117    }
118}

120impl std::fmt::Display for CModuleLoader {
121    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
122        match self.reference().to_string {
123            Some(to_string) => CWrite::new(f).with(|w| to_string(self.payload(), w.into())),
124            None => write!(f, "CModuleLoader-{:?}", self.payload),
125        }
126    }
127}
128
129impl std::fmt::Debug for CModuleLoader {
130    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
131        std::fmt::Display::fmt(self, f)
132    }
133}

135impl ModuleLoader for CModuleLoader {
136    fn load(&self, context: &mut RunContext) -> Result<(), String> {
137        let mut err = String::new();
138        let rc = (self.reference().load.unwrap())(self.payload(), &mut context.into(), (&mut CWrite::new(&mut err)).into());
139        self.result(rc, err)
140    }
141
142    fn load_tokens(&self, target: &MettaMod, metta: Metta) -> Result<(), String> {
143        match self.reference().load_tokens {
144            Some(load_tokens) => {
145                let mut err = String::new();
146                let rc = load_tokens(self.payload(), target.into(), metta.into(), (&mut CWrite::new(&mut err)).into());
147                self.result(rc, err)
148            },
149            None => Ok(()),
150        }
151    }
152}

264/// @ingroup module_group

266///    thread, and may be called concurrently.  Therefore these functions must be fully reentrant.
267///
268#[repr(C)]
269pub struct fs_module_format_t {
270
271    /// @brief Constructs a path for a module with a given name that resides in a parent directory

285

290    /// @param[out]  mod_loader  A module loader instance owned by caller

297

332//QUESTION: What should our multi-thread cross-language API look like?  (I am seeing this as establishing
333//  a pattern we can also use for Grounded Atoms, Spaces, etc.)
334//
335//I see several possible approaches:
336// 1.) Let the C implementations ensure their own concurrency.  This preserves the option for optimal speed,
337//  but might invite problems with badly-behaved extensions.  Especially for host languages where the
338//  interpreter is bound to a specific thread such as Python or JavaScript.
339// 2.) Implement a single-threaded API where we transact all interactions with the host language through
340//  queues to a single interface thread.  This is less rope for the user to hang themselves, but will be
341//  a fundamental bottleneck, so I'd prefer not to go this route
342//
343//My strong opinion is we leave the C API unlimited (ie. Option 1).  Then we can implement the queueing
344//  & synchronization for Python within the hyperonpy layer.  My preference would be to undertake
345//  https://github.com/trueagi-io/hyperon-experimental/issues/283 before trying to implement the queueing &
346//  synchronization layer.  My reasoning is that the bug-surface-area will be a lot smaller in Rust than in C++
347unsafe impl Send for CFsModuleFormat {}
348unsafe impl Sync for CFsModuleFormat {}

350impl FsModuleFormat for CFsModuleFormat {
351    fn paths_for_name(&self, parent_dir: &Path, mod_name: &str) -> Vec<PathBuf> {
352        let parent_dir_c_string = str_as_cstr(parent_dir.to_str().unwrap());
353        let mod_name_c_string = str_as_cstr(mod_name);
354        const BUF_SIZE: usize = 512;
355        let mut buffer = [0 as c_char; BUF_SIZE];
356
357        let bytes_written = (self.reference().path_for_name)(
358            self.payload(),
359            parent_dir_c_string.as_ptr(),
360            mod_name_c_string.as_ptr(),
361            buffer.as_mut_ptr(),
362            BUF_SIZE
363        );
364        if bytes_written > 0 {
365            vec![PathBuf::from(cstr_as_str(buffer[0..=bytes_written].as_ptr()))]
366        } else {
367            vec![]
368        }
369    }
370    fn try_path(&self, path: &Path, mod_name: Option<&str>) -> Option<(Box<dyn ModuleLoader>, ModuleDescriptor)> {
371        let path_c_string = str_as_cstr(path.to_str().unwrap());
372        let mod_name = match mod_name {
373            Some(mod_name) => mod_name,
374            None => path.file_stem().unwrap().to_str().unwrap()
375        };
376        let mod_name_c_string = str_as_cstr(mod_name);
377
378        let mut mod_loader: *const module_loader_t = std::ptr::null(); 
379        let mut mod_descriptor: module_descriptor_t = Default::default();
380        let found = (self.reference().try_path)(self.payload(), path_c_string.as_ptr(), mod_name_c_string.as_ptr(), &mut mod_loader, &mut mod_descriptor);
381        if found {
382            let descriptor = mod_descriptor.into_inner();
383            let loader = CModuleLoader::new(mod_loader);
384            Some((Box::new(loader), descriptor))
385        } else {
386            None
387        }
388    }
