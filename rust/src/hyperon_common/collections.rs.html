18pub struct ListMap<K, V, E: Equality<K> = DefaultEquality>
19{
20    list: Vec<(K, V)>,
21    _phantom: std::marker::PhantomData<E>,
22}

24pub enum ListMapEntry<'a, K, V, E: Equality<K>> {
25    Occupied(K, &'a mut ListMap<K, V, E>),
26    Vacant(K, &'a mut ListMap<K, V, E>),
27}
28
29impl<'a, K, V, E: Equality<K>> ListMapEntry<'a, K, V, E> {
30    pub fn or_default(self) -> &'a mut V where V: Default {
31        self.or_insert(Default::default())
32    }
33
34    pub fn or_insert(self, default: V) -> &'a mut V {
35        match self {
36            ListMapEntry::Occupied(key, map) => map.get_mut(&key).unwrap(),
37            ListMapEntry::Vacant(key, map) => map.insert_entry(key, default),
38        }
39    }
40}

42macro_rules! list_map_iterator {
43    ( $ListMapIter:ident, $Iter:ident, {$( $mut_:tt )?} ) => {
44        pub struct $ListMapIter<'a, K, V> {
45            delegate: std::slice::$Iter<'a, (K, V)>,
46        }
47
48        impl<'a, K, V> Iterator for $ListMapIter<'a, K, V> {
49            type Item = (&'a K, &'a $( $mut_ )? V);
50            fn next(&mut self) -> Option<Self::Item> {
51                match self.delegate.next() {
52                    Some((key, value)) => Some((key, value)),
53                    None => None,
54                }
55            }
56        }
57    }
58}
59
60list_map_iterator!(ListMapIter, Iter, {});
61list_map_iterator!(ListMapIterMut, IterMut, { mut });

63type ListMapIntoIter<K, V> = std::vec::IntoIter<(K, V)>;

65macro_rules! list_map_get {
66    ($get:ident, {$( $mut_:tt )?}) => {
67        pub fn $get(& $( $mut_ )? self, key: &K) -> Option<& $( $mut_ )? V> {
68            for (k, v) in & $( $mut_ )? self.list {
69                if E::eq(k, key) {
70                    return Some(v)
71                }
72            }
73            None
74        }
75    }
76}

78impl<K, V, E: Equality<K>> ListMap<K, V, E> {
79    pub fn new() -> Self {
80        Self{ list: vec![], _phantom: std::marker::PhantomData }
81    }
82
83    pub fn entry<'a>(&'a mut self, key: K) -> ListMapEntry<'a, K, V, E> {
84        match self.get_mut(&key) {
85            Some(_) => ListMapEntry::Occupied(key, self),
86            None => ListMapEntry::Vacant(key, self)
87        }
88    }
89
90    list_map_get!(get, {});
91    list_map_get!(get_mut, {mut});
92
93    pub fn insert(&mut self, key: K, value: V) {
94        self.insert_entry(key, value);
95    }
96
97    fn insert_entry(&mut self, key: K, value: V) -> &mut V {
98        self.list.push((key, value));
99        &mut self.list.last_mut().unwrap().1
100    }
101
102    pub fn is_empty(&self) -> bool {
103        self.list.is_empty()
104    }
105
106    pub fn clear(&mut self) {
107        self.list.clear()
108    }
109
110    pub fn iter(&self) -> ListMapIter<'_, K, V> {
111        ListMapIter{ delegate: self.list.iter() }
112    }
113
114    pub fn iter_mut(&mut self) -> ListMapIterMut<'_, K, V> {
115        ListMapIterMut{ delegate: self.list.iter_mut() }
116    }
117
118    pub fn into_iter(self) -> ListMapIntoIter<K, V> {
119        self.list.into_iter()
120    }
121}

123impl<K, V: PartialEq, E: Equality<K>> PartialEq for ListMap<K, V, E> {
124    fn eq(&self, other: &Self) -> bool {
125        fn left_includes_right<K, V: PartialEq, E: Equality<K>>(left: &ListMap<K, V, E>, right: &ListMap<K, V, E>) -> bool {
126            for e in right.iter() {
127                if left.get(e.0) != Some(e.1) {
128                    return false;
129                }
130            }
131            true
132        }
133        left_includes_right(self, other) && left_includes_right(other, self)
134    }
135}

156pub enum CowArray<T: 'static> {
157    Allocated(Vec<T>),
158    Literal(&'static [T]),
159}
160
161impl<T: 'static> CowArray<T> {
162
163    pub fn new() -> Self {
164        Self::Literal(&[])
165    }
166
167    pub fn as_slice(&self) -> &[T] {
168        match self {
169            Self::Allocated(array) => &*array,
170            Self::Literal(array) => array,
171        }
172    }
173
174    pub fn as_vec_mut(&mut self) -> &mut Vec<T> where T: Clone {
175        match self {
176            Self::Allocated(array) => array,
177            Self::Literal(array) => {
178                *self = Self::Allocated((*array).into());
179                self.as_vec_mut()
180            }
181        }
182    }
183
184    pub fn len(&self) -> usize {
185        self.as_slice().len()
186    }
187
188    pub fn iter(&self) -> impl Iterator<Item=&T> {
189        self.as_slice().iter()
190    }
191}
192
193impl<T: PartialEq> PartialEq for CowArray<T> {
194    fn eq(&self, other: &Self) -> bool {
195        self.as_slice() == other.as_slice()
196    }
197}

201impl<T: Display> Display for CowArray<T> {
202    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
203        write!(f, "[")
204            .and_then(|_| self.as_slice().iter().take(1).fold(Ok(()),
205                |res, atom| res.and_then(|_| write!(f, "{}", atom))))
206            .and_then(|_| self.as_slice().iter().skip(1).fold(Ok(()),
207                |res, atom| res.and_then(|_| write!(f, " {}", atom))))
208            .and_then(|_| write!(f, "]"))
209    }
210}

269/// Helper function to implement Display for all mapping like code structures.
270/// Displays iterator over pairs in a format { <key>: <value>, ... }
271pub fn write_mapping<A, B, I>(f: &mut std::fmt::Formatter, it: I) -> std::fmt::Result
272    where
273        A: Display,
274        B: Display,
275        I: Iterator<Item=(A, B)>
276{
277    write!(f, "{{").and_then(|()| {
278        it.fold((Ok(()), true), |(res, start), (a, b)| {
279            let comma = if start { "" } else { "," };
280            (res.and_then(|()| write!(f, "{} {}: {}", comma, a, b)), false)
281        }).0
282    }).and_then(|()| write!(f, " }}"))
283}
