3/// Serial module defines an API to implement serialization/deserialization of the
4/// grounded atoms. The serialization API can be used for saving grounded atoms to
5/// disk, sending them over network or implement value conversion between
6/// different runtimes (Rust and Python for instance).
7///
8/// One should keep in mind that different runtimes has different set of the
9/// native types for serialization. At the same time the serialization protocol
10/// which is implemented by serializer/deserializer defines how these basic
11/// types are converted into a binary blob. In particular core
12/// library implements a conversion protocol which defines how Python values
13/// are converted into the Rust ones and vice versa. Using native types
14/// instead of using an universal set of types (for example MeTTa stdlib types)
15/// eliminates additional conversion from a native type to a MeTTa one.
16
17/// Trait to implement Rust grounded value serializer. It is not necessary to
18/// implement all methods. By default methods return [Error::NotSupported].
19/// This means that this implementation doesn't support serializing values of
20/// such type.
21pub trait Serializer {
22    /// Serialize bool value.
23    fn serialize_bool(&mut self, _v: bool) -> Result { Err(Error::NotSupported) }
24    /// Serialize i64 value.
25    fn serialize_i64(&mut self, _v: i64) -> Result { Err(Error::NotSupported) }
26    /// Serialize f64 value.
27    fn serialize_f64(&mut self, _v: f64) -> Result { Err(Error::NotSupported) }
28    /// Serialize string value.
29    fn serialize_str(&mut self, _v: &str) -> Result { Err(Error::NotSupported) }
30}
31
32/// Serialization error code
33#[derive(PartialEq, Debug)]
34pub enum Error {
35    /// Serialization of the type is not supported by serializer.
36    NotSupported,
37}
38
39/// Serialization result type
40pub type Result = std::result::Result<(), Error>;
41
42trait PrivHasher : Hasher {}
43impl PrivHasher for DefaultHasher {}
44
45// there are much speedier hashers, but not sure if it's worth the extra dependency given the other options
46impl<H: PrivHasher> Serializer for H {
47    fn serialize_bool(&mut self, v: bool) -> Result { Ok(self.write_u8(v as u8)) }
48    fn serialize_i64(&mut self, v: i64) -> Result { Ok(self.write_i64(v)) }
49    fn serialize_f64(&mut self, v: f64) -> Result { Ok(self.write_i64(f64::to_bits(v).cast_signed())) }
50    fn serialize_str(&mut self, v: &str) -> Result { Ok(v.bytes().for_each(|b| self.write_u8(b))) }
51}
52
53// for debugging
54impl Serializer for String {
55    fn serialize_bool(&mut self, v: bool) -> Result { Ok(self.push_str(&*v.to_string())) }
56    fn serialize_i64(&mut self, v: i64) -> Result { Ok(self.push_str(&*v.to_string())) }
57    fn serialize_f64(&mut self, v: f64) -> Result { Ok(self.push_str(&*v.to_string())) }
58    fn serialize_str(&mut self, v: &str) -> Result { Ok(self.push_str(v)) }
59}
60
61// for speed, but is technically unsafe at usage site because not a valid utf-8 string
62impl Serializer for Vec<u8> {
63    fn serialize_bool(&mut self, v: bool) -> Result { Ok(self.push(v as u8)) }
64    fn serialize_i64(&mut self, v: i64) -> Result { Ok(self.extend(v.to_le_bytes())) }
65    fn serialize_f64(&mut self, v: f64) -> Result { Ok(self.extend(v.to_le_bytes())) }
66    fn serialize_str(&mut self, v: &str) -> Result { Ok(self.extend(v.bytes())) }
67}

69#[derive(Default)]
70pub struct NullSerializer();
71
72impl Serializer for NullSerializer {
73    fn serialize_bool(&mut self, _v: bool) -> Result { Ok(()) }
74    fn serialize_i64(&mut self, _v: i64) -> Result { Ok(()) }
75    fn serialize_f64(&mut self, _v: f64) -> Result { Ok(()) }
76    fn serialize_str(&mut self, _v: &str) -> Result { Ok(()) }
77}

128        let mut hasher = DefaultHasher::new();
