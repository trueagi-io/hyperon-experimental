10/// Grounded function abstraction.
11pub trait GroundedFunction {
12    fn execute(&self, args: &[Atom]) -> Result<Vec<Atom>, ExecError>;
13}
14
15impl<T: Fn(&[Atom]) -> Result<Vec<Atom>, ExecError>>  GroundedFunction for T {
16    fn execute(&self, args: &[Atom]) -> Result<Vec<Atom>, ExecError> {
17        (*self)(args)
18    }
19}
20
21/// Structure to wrap [GroundedFunction] instance to make an [Atom::Grounded].
22pub struct GroundedFunctionAtom<T: GroundedFunction>(Rc<GroundedFunctionAtomContent<T>>);
23
24struct GroundedFunctionAtomContent<T: GroundedFunction> {
25    name: ImmutableString,
26    typ: Atom,
27    func: T,
28}
29
30impl<T: GroundedFunction> GroundedFunctionAtom<T> {
31    /// Constructs new [GroundedFunctionAtom] instance.
32    /// Name also is used to register grounded atom token, see [MettaMod::register_method].
33    pub fn new(name: ImmutableString, typ: Atom, func: T) -> Self {
34        Self(Rc::new(GroundedFunctionAtomContent{ name, typ, func }))
35    }
36
37    /// Returns name of the grounded function to register it as a token.
38    pub fn name(&self) -> &str {
39        self.0.name.as_str()
40    }
41}
42
43impl<T: GroundedFunction> PartialEq for GroundedFunctionAtom<T> {
44    fn eq(&self, _other: &Self) -> bool {
45        true
46    }
47}
48
49impl<T: GroundedFunction> std::fmt::Display for GroundedFunctionAtom<T> {
50    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
51        write!(f, "{}", self.0.name)
52    }
53}
54
55impl<T: GroundedFunction> std::fmt::Debug for GroundedFunctionAtom<T> {
56    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
57        write!(f, "GroundedFunctionAtom[name={}, typ={:?}]", self.0.name, self.0.typ)
58    }
59}
60
61impl<T: GroundedFunction> Clone for GroundedFunctionAtom<T> {
62    fn clone(&self) -> Self {
63        Self(self.0.clone())
64    }
65}
66
67impl<T: GroundedFunction> Grounded for GroundedFunctionAtom<T> {
68    fn type_(&self) -> Atom {
69        self.0.typ.clone()
70    }
71
72    fn as_execute(&self) -> Option<&dyn CustomExecute> {
73        Some(self)

75}

77impl<T: GroundedFunction> CustomExecute for GroundedFunctionAtom<T> {
78    fn execute(&self, args: &[Atom]) -> Result<Vec<Atom>, ExecError> {

80    }

87/// Compares two grounded atoms for equality
88pub fn gnd_eq(a: &dyn GroundedAtom, b: &dyn GroundedAtom) -> bool {
89    // TODO: this function is a hack which is introduced for embedded grounded
90    // types only. It should be replaced by some mechanism which allows defining
91    // a single grounded atom equality procedure in a MeTTa module. This could
92    // be done via defining a type class or equality function overloading.
93    if a.eq_gnd(b) {
94        return true
95    }
96    if a.type_() != b.type_() {
97        return false
98    }
99    if a.type_() == ATOM_TYPE_STRING {
100        Str::try_from(a).unwrap() == Str::try_from(b).unwrap()
101    } else if a.type_() == ATOM_TYPE_NUMBER {
102        Number::try_from(a).unwrap() == Number::try_from(b).unwrap()
103    } else if a.type_() == ATOM_TYPE_BOOL {
104        Bool::try_from(a).unwrap() == Bool::try_from(b).unwrap()
105    } else {
106        false
107    }
