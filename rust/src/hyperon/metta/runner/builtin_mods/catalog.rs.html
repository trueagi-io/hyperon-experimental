10//DISCUSSION: We want to expose more of the pkg_mgmt / catalog system to MeTTa through programmatic
11// interfaces, but the details are unclear.  Most importantly, the use cases are unclear, and those
12// will inform all design decisions.
13//
14//## Potential wish-list for operations available directly in MeTTa:
15// - Ability to work with semver version objects and requirements.  ie. parse versions and requirments
16//    from strings, compare versions to other versions and reqs, etc.
17// - Ability to resolve a module by name (and optionally with a version requirement) without loading it,
18//    returning a ModuleDescriptor and the Catalog in which it was found
19// - Ability to query within a specific catalog, rather than searching all catalogs in priority order
20// - Ability to load an exact module from a specific catalog based on its ModuleDescriptor
21// - Accessor for a resource (see [ResourceKey]) of a module
22// - Ability to inspect / traverse a module dependency hierarchy
23// - Full control over what is "installed".  (inspect, upgrade, install, remove)  See below.
24//
25//## The concept of "installed"
26// The user-level concept of an "installed" module has a imprecise mapping to concepts in the pkg_mgmt /
27// catalog system.  The loosest idea of "installed" is "available", but most users would not consider 
28// software that's freely available on the internet to be "installed".  Most users probably feel like
29// "installed" implies that a specific version of the module is local in their file system.
30//
31// Therefore, modules in remote catalogs wouldn't count as installed, while modules in local catalogs
32// would.  Right now, there are 2 kinds of "local" catalogs.  DirCatalog, and LocalCatalog.
33//
34// DirCatalogs cannot be managed via MeTTa because it is a "read-only" interface over a directory.  We
35// wouldn't want the user to do something that deletes items from a directory the user is managing
36// themselves or under the control of another piece of software like pip or apt.  Additionally, there
37// is no metadata available in a DirCatalog to automatically upgrade any of the contents.
38//
39// LocalCatalogs are designed to be managed by Hyperon, from MeTTa.  A LocalCatalog mirrors one or more
40// upstream module sources (remote Catalogs), so it's expected that a user will install, upgrade, remove,
41// etc. modules in a LocalCatalog. 
42//
43// Therefore, an interface for "managing" what is installed must be limited to LocalCatalogs.
44//
45//## Should it be supported to list the contents of a DirCatalog?
46// A DirCatalog that contains only MeTTa modules, such as the 'modules' directory in the MeTTa config dir,
47//    can be listed easily enough.  Things get complicated however because any directory can be loaded as
48//    a MeTTa module.  Therefore when configuration adds a heterogeneous directory, such as `site-packages`,
49//    the DirCatalog will report every subdir as a module.  Which is correct, but probably not useful.
50//    At the very least, it makes `!(catalog-list! all)` much more noisy
51//
52
53pub static CATALOG_METTA: &'static str = include_str!("catalog.metta");
54
55/// Loader to Initialize the "catalog" module
56#[derive(Debug)]
57pub(crate) struct CatalogModLoader;
58
59impl ModuleLoader for CatalogModLoader {
60    fn load(&self, context: &mut RunContext) -> Result<(), String> {
61        let space = GroundingSpace::new();
62        context.init_self_module(space.into(), None);
63        let _ = self.load_tokens(context.module(), context.metta.clone());
64        let parser = SExprParser::new(CATALOG_METTA);
65        context.push_parser(Box::new(parser));
66        Ok(())
67    }
68
69    fn load_tokens(&self, target: &MettaMod, metta: Metta) -> Result<(), String> {
70        let mut tref = target.tokenizer().borrow_mut();
71
72        let catalog_list_op = Atom::gnd(CatalogListOp::new(metta.clone()));
73        tref.register_token(regex(r"catalog-list!"), move |_| { catalog_list_op.clone() });
74        let catalog_update_op = Atom::gnd(CatalogUpdateOp::new(metta.clone()));
75        tref.register_token(regex(r"catalog-update!"), move |_| { catalog_update_op.clone() });
76        let catalog_clear_op = Atom::gnd(CatalogClearOp::new(metta.clone()));
77        tref.register_token(regex(r"catalog-clear!"), move |_| { catalog_clear_op.clone() });
78
79        Ok(())
80    }
81}

83/// Lists contents of all Catalogs that support the "list" method
84#[derive(Clone, Debug)]
85pub struct CatalogListOp {
86    metta: Metta
87}
88
89grounded_op!(CatalogListOp, "catalog-list!");
90
91impl CatalogListOp {
92    pub fn new(metta: Metta) -> Self {
93        Self{ metta }
94    }
95}
96
97impl Grounded for CatalogListOp {
98    fn type_(&self) -> Atom {
99        //TODO-FUTURE, we may want to return the list as atoms, but now it just prints to stdout
100        Atom::expr([ARROW_SYMBOL, ATOM_TYPE_SYMBOL, UNIT_TYPE])
101    }
102
103    fn as_execute(&self) -> Option<&dyn CustomExecute> {
104        Some(self)
105    }
106}
107
108impl CustomExecute for CatalogListOp {
109    fn execute(&self, args: &[Atom]) -> Result<Vec<Atom>, ExecError> {
110        let arg_error = "catalog-list! expects a catalog name, or \"all\" to list all available";
111        let cat_name_arg_atom = args.get(0).ok_or_else(|| ExecError::from(arg_error))?;
112        let cat_name = if let Atom::Symbol(cat_name) = cat_name_arg_atom {
113            cat_name.name()
114        } else {
115            return Err(ExecError::from(arg_error));
116        };
117
118        fn list_catalog(cat: &dyn crate::metta::runner::ModuleCatalog) {
119            if let Some(cat_iter) = cat.list() {
120                println!("{}:", cat.display_name());
121                for desc in cat_iter {
122                    println!("   {desc}");
123                }
124            }
125        }
126
127        let mut found_one = false;
128        if cat_name == "all" || cat_name == "specified-mods" {
129            if let Some(specified_mods) = &self.metta.environment().specified_mods {
130                list_catalog(specified_mods);
131                found_one = true;
132            }
133        }
134        for cat in self.metta.environment().catalogs() {
135            if cat_name == "all" || cat_name == cat.display_name() {
136                list_catalog(cat);
137                found_one = true;
138            }
139        }
140
141        if !found_one {
142            return Err(ExecError::from(format!("no catalog(s) identified by \"{cat_name}\"")));
143        }
144        unit_result()
145    }
146}

169        Some(self)

173impl CustomExecute for CatalogUpdateOp {
174    fn execute(&self, args: &[Atom]) -> Result<Vec<Atom>, ExecError> {
175        let arg_error = "catalog-update! expects a catalog name, or \"all\" to update all";
176        let cat_name_arg_atom = args.get(0).ok_or_else(|| ExecError::from(arg_error))?;
177        let cat_name = if let Atom::Symbol(cat_name) = cat_name_arg_atom {
178            cat_name.name()
179        } else {
180            return Err(ExecError::from(arg_error));
181        };
182
183        let mut found_one = false;
184        if cat_name == "all" || cat_name == "specified-mods" {
185            if let Some(specified_mods) = &self.metta.environment().specified_mods {
186                specified_mods.fetch_newest_for_all(UpdateMode::FetchLatest)?;
187                found_one = true;
188            }
189        }
190
191        for cat in self.metta.environment().catalogs() {
192            match cat.as_managed() {
193                Some(cat) => if cat_name == "all" || cat_name == cat.display_name() {
194                    cat.fetch_newest_for_all(UpdateMode::FetchLatest)?;
195                    found_one = true;
196                },
197                None => {}
198            }
199        }
200
201        if !found_one {
202            return Err(ExecError::from(format!("no catalog(s) identified by \"{cat_name}\"")));
203        }
204        unit_result()
205    }
206}
