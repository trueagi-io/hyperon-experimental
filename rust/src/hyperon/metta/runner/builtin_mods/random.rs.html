27#[derive(Clone, Debug)]
28pub struct RandomGenerator(Rc<RefCell<StdRng>>);
29
30impl RandomGenerator {
31    fn from_os_rng() -> Self {
32        Self(Rc::new(RefCell::new(StdRng::from_os_rng())))
33    }
34
35    fn from_seed_u64(seed: u64) -> Self {
36        Self(Rc::new(RefCell::new(StdRng::seed_from_u64(seed))))
37    }
38
39    fn reseed_from_u64(&self, seed: u64) {
40        *self.0.borrow_mut() = StdRng::seed_from_u64(seed);
41    }
42
43    fn reset(&self) {
44        *self.0.borrow_mut() = StdRng::from_os_rng();
45    }
46
47    fn random_range<T, R>(&self, range: R) -> T 
48        where
49            T: rand::distr::uniform::SampleUniform,
50            R: rand::distr::uniform::SampleRange<T>,
51    {
52        self.0.borrow_mut().random_range(range)
53    }
54}
55
56impl Grounded for RandomGenerator {
57    fn type_(&self) -> Atom {
58        ATOM_TYPE_RANDOM_GENERATOR 
59    }
60}
61
62impl Display for RandomGenerator {
63    fn fmt(&self, f: &mut Formatter) -> std::fmt::Result {
64        write!(f, "RandomGenerator-{:?}", self.0.as_ptr())
65    }
66}
67
68impl PartialEq for RandomGenerator {
69    fn eq(&self, other: &Self) -> bool {
70        self.0.as_ptr() == other.0.as_ptr()
71    }
72}

75#[derive(Debug)]
76pub(crate) struct RandomModLoader;
77
78impl ModuleLoader for RandomModLoader {
79    fn load(&self, context: &mut RunContext) -> Result<(), String> {
80        // Initialize module's space
81        let space = GroundingSpace::new();
82        context.init_self_module(space.into(), None);
83
84        // Load module's tokens
85        let _ = self.load_tokens(context.module(), context.metta.clone())?;
86
87        // Parse MeTTa code of the module
88        let parser = SExprParser::new(RANDOM_METTA);
89        context.push_parser(Box::new(parser));
90
91        Ok(())
92    }
93
94    fn load_tokens(&self, target: &MettaMod, _metta: Metta) -> Result<(), String> {
95        let mut tref = target.tokenizer().borrow_mut();
96
97        tref.register_function(GroundedFunctionAtom::new(
98            r"random-int".into(),
99            Atom::expr([ARROW_SYMBOL, ATOM_TYPE_RANDOM_GENERATOR, ATOM_TYPE_NUMBER, ATOM_TYPE_NUMBER, ATOM_TYPE_NUMBER]),
100            random_int));
101
102        tref.register_function(GroundedFunctionAtom::new(
103            r"random-float".into(),
104            Atom::expr([ARROW_SYMBOL, ATOM_TYPE_RANDOM_GENERATOR, ATOM_TYPE_NUMBER, ATOM_TYPE_NUMBER, ATOM_TYPE_NUMBER]),
105            random_float));
106
107        tref.register_function(GroundedFunctionAtom::new(
108            r"set-random-seed".into(),
109            Atom::expr([ARROW_SYMBOL, ATOM_TYPE_RANDOM_GENERATOR, ATOM_TYPE_NUMBER, UNIT_TYPE]),
110            set_random_seed));
111
112        tref.register_function(GroundedFunctionAtom::new(
113            r"new-random-generator".into(),
114            Atom::expr([ARROW_SYMBOL, ATOM_TYPE_NUMBER, ATOM_TYPE_RANDOM_GENERATOR]),
115            new_random_generator));
116
117        tref.register_function(GroundedFunctionAtom::new(
118            r"reset-random-generator".into(),
119            Atom::expr([ARROW_SYMBOL, ATOM_TYPE_RANDOM_GENERATOR, UNIT_ATOM]),
120            reset_random_generator));
121
122        tref.register_function(GroundedFunctionAtom::new(
123            r"flip".into(),
124            Atom::expr([ARROW_SYMBOL, ATOM_TYPE_BOOL]),
125            flip));
126
127        let generator = Atom::gnd(RandomGenerator::from_os_rng());
128        tref.register_token(Regex::new(r"&rng").unwrap(),move |_| { generator.clone() });
129
130        Ok(())
131    }
132}

135fn random_int(args: &[Atom]) -> Result<Vec<Atom>, ExecError> {
136    let arg_error = || ExecError::from("random-int expects three arguments: random generator, number (start) and number (end)");
137    let generator = args.get(0).ok_or_else(arg_error)?.into();
138    let start: i64 = args.get(1).and_then(Number::from_atom).ok_or_else(arg_error)?.into();
139    let end: i64 = args.get(2).and_then(Number::from_atom).ok_or_else(arg_error)?.into();
140    let generator = Atom::as_gnd::<RandomGenerator>(generator).ok_or("random-int expects a random generator as its argument")?;
141    let range = start..end;
142    if range.is_empty() {
143        return Err(ExecError::from("RangeIsEmpty"));
144    }
145    Ok(vec![Atom::gnd(Number::Integer(generator.random_range(range)))])
146}

148fn random_float(args: &[Atom]) -> Result<Vec<Atom>, ExecError> {
149    let arg_error = || ExecError::from("random-float expects three arguments: random generator, number (start) and number (end)");
150    let generator = args.get(0).ok_or_else(arg_error)?.into();
151    let start: f64 = args.get(1).and_then(Number::from_atom).ok_or_else(arg_error)?.into();
152    let end: f64 = args.get(2).and_then(Number::from_atom).ok_or_else(arg_error)?.into();
153    let generator = Atom::as_gnd::<RandomGenerator>(generator).ok_or("random-float expects a random generator as its argument")?;
154    let range = start..end;
155    if range.is_empty() {
156        return Err(ExecError::from("RangeIsEmpty"));
157    }
158    Ok(vec![Atom::gnd(Number::Float(generator.random_range(range)))])
159}

161fn set_random_seed(args: &[Atom]) -> Result<Vec<Atom>, ExecError> {
162    let arg_error = || ExecError::from("set-random-seed expects two arguments: random generator and number (seed)");
163    let generator = args.get(0).ok_or_else(arg_error)?.into();
164    let seed: i64 = args.get(1).and_then(Number::from_atom).ok_or_else(arg_error)?.into();
165    let generator = Atom::as_gnd::<RandomGenerator>(generator).ok_or("set-random-seed expects a random generator as its argument")?;
166    generator.reseed_from_u64(seed as u64);
167    Ok(vec![UNIT_ATOM])
168}
169
170fn new_random_generator(args: &[Atom]) -> Result<Vec<Atom>, ExecError> {
171    let arg_error = || ExecError::from("new-random-generator expects one argument: number (seed)");
172    let seed: i64 = args.get(0).and_then(Number::from_atom).ok_or_else(arg_error)?.into();
173    let new_generator = RandomGenerator::from_seed_u64(seed as u64);
174    Ok(vec![Atom::gnd(new_generator)])
175}
176
177fn reset_random_generator(args: &[Atom]) -> Result<Vec<Atom>, ExecError> {
178    let arg_error = || ExecError::from("reset-random-generator expects one argument: random generator");
179    let generator = args.get(0).ok_or_else(arg_error)?.into();
180    let generator = Atom::as_gnd::<RandomGenerator>(generator).ok_or("set-random-seed expects a random generator as its argument")?;
181    generator.reset();
182    Ok(vec![UNIT_ATOM])
183}

264    #[test]
265    fn random_op() {
266        let res = random_int(&mut vec![expr!({RandomGenerator::from_os_rng()}), expr!({Number::Integer(0)}), expr!({Number::Integer(5)})]);
267        let range = 0..5;
268        let res_i64: i64 = res.unwrap().get(0).and_then(Number::from_atom).unwrap().into();
269        assert!(range.contains(&res_i64));
270        let res = random_int(&mut vec![expr!({RandomGenerator::from_os_rng()}), expr!({Number::Integer(2)}), expr!({Number::Integer(-2)})]);
271        assert_eq!(res, Err(ExecError::from("RangeIsEmpty")));
272
273        let res = random_float(&mut vec![expr!({RandomGenerator::from_os_rng()}), expr!({Number::Integer(0)}), expr!({Number::Integer(5)})]);
274        let range = 0.0..5.0;
275        let res_f64: f64 = res.unwrap().get(0).and_then(Number::from_atom).unwrap().into();
276        assert!(range.contains(&res_f64));
277        let res = random_float(&mut vec![expr!({RandomGenerator::from_os_rng()}), expr!({Number::Integer(0)}), expr!({Number::Integer(0)})]);
278        assert_eq!(res, Err(ExecError::from("RangeIsEmpty")));
279
280        let gen = new_random_generator(&mut vec![expr!({Number::Integer(0)})]);
281        let res1 = random_float(&mut vec![expr!({gen}), expr!({Number::Integer(0)}), expr!({Number::Integer(5)})]);
282        let gen = new_random_generator(&mut vec![expr!({Number::Integer(0)})]);
283        let res2 = random_float(&mut vec![expr!({gen}), expr!({Number::Integer(0)}), expr!({Number::Integer(5)})]);
284        assert_eq!(res1, res2);
285
286        let gen = new_random_generator(&mut vec![expr!({Number::Integer(0)})]);
287        let _ = set_random_seed(&mut vec![expr!({gen.clone()}), expr!({Number::Integer(0)})]);
288        let res1 = random_float(&mut vec![expr!({gen.clone()}), expr!({Number::Integer(0)}), expr!({Number::Integer(5)})]);
289        let _ = set_random_seed(&mut vec![expr!({gen.clone()}), expr!({Number::Integer(5)})]);
290        let res2 = random_float(&mut vec![expr!({gen.clone()}), expr!({Number::Integer(0)}), expr!({Number::Integer(5)})]);
291        assert_eq!(res1, res2);
292    }
