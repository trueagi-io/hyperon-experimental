16/// A set of keys describing how to access a module via git.  Deserialized from within a [PkgInfo]
17///  or a catalog file [CatalogFileFormat]
18#[derive(Clone, Debug, Default, Serialize, Deserialize)]
19pub struct ModuleGitLocation {
20    /// Indicates that the dependency module should be fetched from the specified `git` URL
21    #[serde(default)]
22    pub git_url: Option<String>,
23
24    /// A `git`` branch to fetch.  Will be ignored if `git_url` is `None`.  Uses the repo's
25    /// default branch if left unspecified
26    #[serde(default)]
27    pub git_branch: Option<String>,
28
29    /// A subdirectory within the git repo to use as the module, effectively ignoring the rest
30    /// of the repo contents.  The subdir must be a relative path within the repo.
31    #[serde(default)]
32    pub git_subdir: Option<PathBuf>,
33
34    /// A file within the git repo to use as the module.  The file path must be a relative path
35    /// within the repo or `git_subdir` directory if provided.
36    #[serde(default)]
37    pub git_main_file: Option<PathBuf>,
38
39    /// A path relative to the catalog repo, from which to load the module
40    ///
41    /// WARNING: This key will be ignored if a `git_url` is provided.
42    #[serde(default)]
43    pub local_path: Option<PathBuf>,
44}
45
46impl ModuleGitLocation {
47    /// Fetches the module from git if it doesn't exist in `local_cache_dir`, and then returns
48    /// a ModuleLoader & ModuleDescriptor pair for the module
49    pub(crate) fn fetch_and_get_loader<'a, FmtIter: Iterator<Item=&'a dyn FsModuleFormat>>(&self, fmts: FmtIter, mod_name: &str, local_cache_dir: PathBuf, catalog_file_path: &Path, update_mode: UpdateMode) -> Result<Option<(Box<dyn ModuleLoader>, ModuleDescriptor)>, String> {
50
51        //If a git URL is specified in the entry, see if we have it in the git-cache and
52        // clone it locally if we don't
53        if self.git_url.is_some() {
54            let cached_repo = self.get_cache(mod_name, local_cache_dir)?;
55            cached_repo.update(update_mode)?;
56
57            let mod_path = match &self.git_main_file {
58                Some(main_file) => cached_repo.local_path().join(main_file),
59                None => cached_repo.local_path().to_owned(),
60            };
61            return loader_for_module_at_path(fmts, &mod_path, Some(mod_name), None);
62        }
63
64        //If a `local_path` was specified, then load the module from there
65        match &self.local_path {
66            Some(local_path) => {
67                let mod_path = catalog_file_path.parent().unwrap().join(local_path);
68                return loader_for_module_at_path(fmts, &mod_path, Some(mod_name), None);
69            },
70            None => {}
71        }
72
73        Ok(None)
74    }
75    pub(crate) fn get_cache(&self, mod_name: &str, local_cache_dir: PathBuf) -> Result<CachedRepo, String> {
76        let url = self.git_url.as_ref().unwrap();
77        let branch = self.git_branch.as_ref().map(|s| s.as_str());
78        let subdir = self.git_subdir.as_ref().map(|p| p.as_path());
79        CachedRepo::new(mod_name, local_cache_dir, url, branch, subdir)
80    }
81    pub(crate) fn uid(&self) -> u64 {
82        let subdir_string;
83        let subdir_str = if let Some(p) = &self.git_subdir {
84            subdir_string = format!("{p:?}");
85            subdir_string.as_str()
86        } else {""};
87        let main_file_string;
88        let main_file_str = if let Some(p) = &self.git_main_file {
89            main_file_string = format!("{p:?}");
90            main_file_string.as_str()
91        } else {""};
92        let unique_string = format!("{}-{}-{subdir_str}-{main_file_str}",
93            self.git_url.as_ref().map(|s| s.as_str()).unwrap_or(""),
94            self.git_branch.as_ref().map(|s| s.as_str()).unwrap_or(""),
95        );
96        ModuleDescriptor::uid_from_ident_bytes_and_fmt_id(unique_string.as_bytes(), 0)
97    }
98    /// Returns a new ModuleGitLocation.  This is a convenience; the usual interface involves deserializing this struct
99    pub(crate) fn new(url: String) -> Self {
100        let mut new_self = Self::default();
101        new_self.git_url = Some(url);
102        new_self
103    }
104    pub(crate) fn get_url(&self) -> Option<&str> {
105        self.git_url.as_ref().map(|s| s.as_str())
106    }
107}

109/// Struct that matches the catalog.json file fetched from the `_catalog.repo`
110#[derive(Serialize, Deserialize, Debug, Default)]
111struct CatalogFileFormat {
112    modules: BTreeMap<String, Vec<CatalogFileMod>>
113}
114
115impl CatalogFileFormat {
116    fn find_mods_with_name(&self, name: &str) -> Vec<ModuleDescriptor> {
117        let mut results = vec![];
118        if let Some(cat_mod_vec) = self.modules.get(name) {
119            for cat_mod in cat_mod_vec {
120                let uid = cat_mod.git_location.uid();
121                let descriptor = ModuleDescriptor::new(name.to_string(), cat_mod.version.clone(), Some(uid));
122                results.push(descriptor);
123            }
124        }
125        results
126    }
127    /// Scans the catalog looking for a single module that matches the provided descriptor
128    fn find_mod_with_descriptor(&self, descriptor: &ModuleDescriptor) -> Option<&CatalogFileMod> {
129        if let Some(cat_mod_vec) = self.modules.get(descriptor.name()) {
130            for cat_mod in cat_mod_vec.iter() {
131                if cat_mod.version.as_ref() == descriptor.version() {
132                    //NOTE ModuleDescriptors from GitCatalog always have a uid
133                    if Some(cat_mod.git_location.uid()) == descriptor.uid() {
134                        return Some(cat_mod);
135                    }
136                }
137            }
138        }
139        None
140    }
141    fn add(&mut self, new_mod: CatalogFileMod) -> Result<ModuleDescriptor, String> {
142        let uid = new_mod.git_location.uid();
143        let descriptor = ModuleDescriptor::new(new_mod.name.clone(), new_mod.version.clone(), Some(uid));
144        if self.find_mod_with_descriptor(&descriptor).is_none() {
145            let cat_mod_vec = self.modules.entry(new_mod.name.clone()).or_insert(vec![]);
146            cat_mod_vec.push(new_mod);
147        }
148        Ok(descriptor)
149    }
150}

168/// Provides an interface to a git repo hosting a table of available modules
169#[derive(Debug)]
170pub struct GitCatalog {
171    /// The name of this catalog
172    name: String,
173    /// The FsModuleFormats from the environment, to load the modules from their respective repositories
174    fmts: Arc<Vec<Box<dyn FsModuleFormat>>>,
175    /// An interval in seconds to control the refresh of the catalog from the upstream source
176    refresh_time: u64,
177    /// The git repo for the catalog info.  This is the table-of-contents for the catalog, not the modules
178    catalog_repo: Option<CachedRepo>,
179    /// A path to a directory where this catalog may cache local files.  When a `GitCatalog` is upstream of
180    /// a [LocalCatalog], this directory will not contain modules, but modules will be stored here when the
181    /// `GitCatalog` is used by itself
182    caches_dir: PathBuf,
183    /// The path to the catalog file(s), to store the metadata to connect a module to its source location
184    /// parameters.  This path does not have any reliable connection to the on-disk location of the modules
185    catalog_file_path: PathBuf,
186    /// The in-memory catalog object, mirroring what is on disk
187    catalog: Mutex<Option<CatalogFileFormat>>,
188}
189
190impl GitCatalog {
191    fn new_internal(fmts: Arc<Vec<Box<dyn FsModuleFormat>>>, name: &str, caches_dir: PathBuf, catalog_file_path: PathBuf, catalog: Option<CatalogFileFormat>) -> Self {
192        Self {
193            name: name.to_string(),
194            fmts,
195            refresh_time: 0,
196            catalog_repo: None,
197            caches_dir,
198            catalog_file_path,
199            catalog: Mutex::new(catalog),
200        }
201    }
202    /// Creates a new GitCatalog with the name and url specified.  `refresh_time` is the time, in
203    /// seconds, between refreshes of the catalog file
204    pub fn new(caches_dir: &Path, fmts: Arc<Vec<Box<dyn FsModuleFormat>>>, name: &str, url: &str, refresh_time: u64) -> Result<Self, String> {
205        let caches_dir = caches_dir.join(name);
206        let catalog_repo_dir = caches_dir.join("_catalog.repo");
207        let catalog_repo_name = format!("{name}-catalog.repo");
208        let catalog_repo = CachedRepo::new(&catalog_repo_name, catalog_repo_dir, url, None, None)?;
209        let mut new_self = Self::new_internal(fmts, name, caches_dir, catalog_repo.local_path().join("catalog.json"), None);
210        new_self.refresh_time = refresh_time;
211        new_self.catalog_repo = Some(catalog_repo);
212        Ok(new_self)
213    }
214    /// Used for a git-based catalog that isn't synced to a remote source
215    ///
216    /// The primary use of this method is to make a `GitCatalog` to back the `git-module!` MeTTa operation,
217    /// so it will track the locations of explicitly sourced modules.  It's a fairly special-purpose object
218    /// designed to associate git information (repo url, branch, etc.) with some modules available through a
219    /// [LocalCatalog], so the user can "install" modules by fetching them from wherever, and then pull the
220    /// latest version without needing to re-supply the details about where to fetch the module from.
221    pub fn new_without_source_repo(caches_dir: &Path, fmts: Arc<Vec<Box<dyn FsModuleFormat>>>, name: &str) -> Result<Self, String> {
222        let caches_dir = caches_dir.join(name);
223        let catalog_file_path = caches_dir.join("_catalog.json");
224        let new_self = Self::new_internal(fmts, name, caches_dir, catalog_file_path, Some(CatalogFileFormat::default()));
225        if new_self.catalog_file_path.exists() {
226            new_self.parse_catalog()?
227        } else {
228            new_self.write_catalog()?;
229        }
230        Ok(new_self)
231    }
232    /// Registers a new module in the catalog with a specified remote location, and returns
233    /// the [ModuleDescriptor] to refer to that module
234    ///
235    /// WARNING: This method is incompatible with a catalog synced to an upstream source
236    pub(crate) fn register_mod(&self, mod_name: &str, version: Option<&semver::Version>, git_location: &ModuleGitLocation) -> Result<ModuleDescriptor, String> {
237        if self.catalog_repo.is_some() {
238            return Err(format!("cannot explicitly register module in a catalog synced to an upstream source"));
239        }
240        let descriptor = {
241            let mut catalog_ref = self.catalog.lock().unwrap();
242            catalog_ref.as_mut().unwrap().add(CatalogFileMod::new(mod_name.to_string(), version.cloned(), git_location.clone()))?
243        };
244        self.write_catalog()?;
245        Ok(descriptor)
246    }
247    /// Scans the catalog and finds all the modules with a given name
248    fn find_mods_with_name(&self, name: &str) -> Vec<ModuleDescriptor> {
249        let cat_lock = self.catalog.lock().unwrap();
250        let catalog = cat_lock.as_ref().unwrap();
251        catalog.find_mods_with_name(name)
252    }
253    fn refresh_catalog(&self, update_mode: UpdateMode) -> Result<(), String> {
254        if let Some(catalog_repo) = &self.catalog_repo {
255            //Update the catalog from the git cache
256            let did_update = match catalog_repo.update(update_mode) {
257                Ok(did_update) => did_update,
258                Err(e) => {
259                    log::warn!("Warning: error encountered attempting to fetch remote catalog: {}, {e}", self.name);
260                    false
261                }
262            };
263
264            //Parse the catalog JSON file if we need to
265            if did_update || self.catalog_is_uninit() {
266                self.parse_catalog()?;
267            }
268        }
269        Ok(())
270    }
271    fn catalog_is_uninit(&self) -> bool {
272        let catalog = self.catalog.lock().unwrap();
273        catalog.is_none()
274    }
275    fn parse_catalog(&self) -> Result<(), String> {
276        let mut catalog = self.catalog.lock().unwrap();
277        match File::open(&self.catalog_file_path) {
278            Ok(file) => {
279                let reader = BufReader::new(file);
280                *catalog = Some(serde_json::from_reader(reader).unwrap());
281                Ok(())
282            },
283            Err(e) => {
284                Err(format!("Error reading catalog file. remote catalog unavailable: {}, {e}", self.name))
285            }
286        }
287    }
288    /// Writes the catalog to a file, overwriting the file that is currently on disk
289    fn write_catalog(&self) -> Result<(), String> {
290        let cat_lock = self.catalog.lock().unwrap();
291        let catalog = cat_lock.as_ref().unwrap();
292        let file = File::create(&self.catalog_file_path).map_err(|e| e.to_string())?;
293        let writer = BufWriter::new(file);
294        serde_json::to_writer(writer, catalog).map_err(|e| e.to_string())?;
295        Ok(())
296    }
297}

299impl ModuleCatalog for GitCatalog {
300    fn display_name(&self) -> String {
301        self.name.clone()
302    }
303    fn lookup(&self, name: &str) -> Vec<ModuleDescriptor> {
304        match self.refresh_catalog(UpdateMode::TryFetchIfOlderThan(self.refresh_time)) {
305            Ok(_) => {},
306            Err(e) => {
307                log::warn!("{e}");
308                return vec![]
309            }
310        }
311
312        //Find the modules that match in the catalog
313        self.find_mods_with_name(name)
314    }
315    fn get_loader(&self, descriptor: &ModuleDescriptor) -> Result<Box<dyn ModuleLoader>, String> {
316        self.refresh_catalog(UpdateMode::TryFetchIfOlderThan(self.refresh_time))?;
317
318        let cat_lock = self.catalog.lock().unwrap();
319        let catalog = cat_lock.as_ref().unwrap();
320        let module = catalog.find_mod_with_descriptor(descriptor)
321            .ok_or_else(|| format!("Error: module {descriptor} no longer exists in catalog {}", self.display_name()))?;
322
323        let mod_dir_name = super::managed_catalog::dir_name_from_descriptor(descriptor);
324        let fallback_mod_dir = self.caches_dir.join(mod_dir_name);
325
326        Ok(Box::new(GitModLoader{
327            module: module.clone(),
328            fmts: self.fmts.clone(),
329            fallback_mod_dir,
330            catalog_file_path: self.catalog_file_path.clone(),
331        }))
332    }
333    fn sync_toc(&self, update_mode: UpdateMode) -> Result<(), String> {
334        self.refresh_catalog(update_mode)?;
335        Ok(())
336    }
337    fn as_any(&self) -> Option<&dyn Any> {
338        Some(self as &dyn Any)
339    }
340}

342#[derive(Debug)]
343pub struct GitModLoader {
344    module: CatalogFileMod,
345    fmts: Arc<Vec<Box<dyn FsModuleFormat>>>,
346    /// A local path to clone the module into, if a path isn't provided by a downstream LocalCatalog
347    fallback_mod_dir: PathBuf,
348    /// The location of the catalog file from which the `module` field was deserialized
349    catalog_file_path: PathBuf,
350}
351
352impl ModuleLoader for GitModLoader {
353    fn prepare(&self, local_dir: Option<&Path>, update_mode: UpdateMode) -> Result<Option<Box<dyn ModuleLoader>>, String> {
354        let local_dir = match local_dir {
355            Some(local_dir) => local_dir,
356            None => &self.fallback_mod_dir
357        };
358        let loader = match self.module.git_location.fetch_and_get_loader(self.fmts.iter().map(|f| &**f), &self.module.name, local_dir.to_owned(), &self.catalog_file_path, update_mode)? {
359            Some((loader, _)) => loader,
360            None => unreachable!(),
361        };
362        Ok(Some(loader))
363    }
364    fn load(&self, _context: &mut RunContext) -> Result<(), String> {
365        unreachable!()
366    }
367}
