22static SIGINT_RECEIVED_COUNT: Mutex<usize> = Mutex::new(0);

24#[derive(Parser)]
25#[command(version, about)]
26struct CliArgs {
27    /// .metta file to execute.  `metta` will run in interactive mode if no file is supplied
28    file: Option<PathBuf>,
29
30    /// Additional include directory paths
31    #[arg(short, long)]
32    include_paths: Vec<PathBuf>,
33}
34
35fn main() -> Result<()> {
36    let cli_args = CliArgs::parse();
37    let _ = env_logger::builder().filter_level(log::LevelFilter::Info).try_init();
38
39    //If we have a metta_file, then the working dir is the parent of that file
40    //If we are running in interactive mode, it's the working dir at the time the repl is invoked
41    let metta_working_dir: PathBuf = match &cli_args.file {
42        Some(metta_file) => {
43            metta_file.parent().unwrap().into()
44        },
45        None => {
46            match std::env::current_dir() {
47                Ok(cwd) => cwd,
48                Err(_) => PathBuf::from("./").canonicalize().unwrap(),
49            }
50        }
51    };
52
53    //Create our MeTTa runtime environment
54    let mut metta = MettaShim::new(metta_working_dir, cli_args.include_paths);
55
56    //Init our runtime environment
57    let repl_params = ReplParams::new(&metta);
58
59    //Spawn a signal handler background thread, to deal with passing interrupts to the execution loop
60    ctrlc::set_handler(move || {
61        let mut signal_received_cnt = SIGINT_RECEIVED_COUNT.lock().unwrap();
62        match *signal_received_cnt {
63            0 => println!("Interrupt received, stopping MeTTa..."),
64            1 => println!("Stopping in progress.  Please wait..."),
65            _ => {
66                println!("Ok, I get it!  Yeesh!");
67                exit(-1);
68            },
69        }
70        *signal_received_cnt += 1;
71        drop(signal_received_cnt);
72    })?;
73
74
75    //If we have .metta files to run, then run them
76    if let Some(metta_file) = &cli_args.file {
77
78        //Only print the output from the primary .metta file
79        let metta_file = std::io::BufReader::new(std::fs::File::open(metta_file)?);
80        metta.exec(metta_file);
81        metta.print_result();
82        Ok(())
83
84    } else {
85
86        //Otherwise enter interactive mode
87        start_interactive_mode(repl_params, metta).map_err(|err| err.into())
88    }
89}

91fn show_welcome_message() {
92    println!("Visit https://metta-lang.dev/ for tutorials.");
93    println!("Execute !(help!) to get list of the standard library functions.");
94}

97// RUST_LOG=rustyline=debug cargo run --example example 2> debug.log
98fn start_interactive_mode(repl_params: ReplParams, mut metta: MettaShim) -> rustyline::Result<()> {
99
100    //Run the built-in repl-init code
101    metta.exec(builtin_init_metta_code().as_str());
102
103    //Run the repl init file
104    if let Some(repl_config_metta_path) = &repl_params.repl_config_metta_path {
105        let init_metta_file = std::io::BufReader::new(std::fs::File::open(repl_config_metta_path)?);
106        metta.exec(init_metta_file);
107    }
108
109    let max_len = metta.get_config_int(CFG_HISTORY_MAX_LEN).unwrap_or_else(|| 500);
110
111    //Init RustyLine
112    let config = Config::builder()
113        .history_ignore_space(true)
114        .max_history_size(max_len as usize).unwrap()
115        .history_ignore_dups(false).unwrap()
116        .completion_type(CompletionType::List)
117        .edit_mode(EditMode::Emacs)
118        .build();
119    let helper = ReplHelper::new(metta);
120    let mut rl = Editor::with_config(config)?;
121
122    //QUESTION: Should we provide a config to use vi key bindings vs. Emacs?
123
124    rl.set_helper(Some(helper));
125    //KEY BEHAVIOR: Enter and ctrl-M will add a newline when the cursor is in the middle of a line, while
126    // ctrl-J will submit the line.
127    //TODO: Document this behavior in the README.md, and possibly other key bindings also
128    rl.bind_sequence(KeyEvent( KeyCode::Enter, Modifiers::NONE ), Cmd::AcceptOrInsertLine {
129        accept_in_the_middle: false,
130    });
131    rl.bind_sequence(KeyEvent::ctrl('j'), EventHandler::Conditional(Box::new(EnterKeyHandler::new(rl.helper().unwrap().force_submit.clone()))));
132    rl.bind_sequence(KeyEvent::alt('n'), Cmd::HistorySearchForward);
133    rl.bind_sequence(KeyEvent::alt('p'), Cmd::HistorySearchBackward);
134    if let Some(history_path) = &repl_params.history_file {
135        if rl.load_history(history_path).is_err() {
136            println!("No previous history found.");
137        }
138    }
139
140    show_welcome_message();
141
142    //The Interpreter Loop
143    loop {
144
145        //Set the prompt based on the MeTTa pragma settings
146        let prompt = {
147            let helper = rl.helper_mut().unwrap();
148            let mut metta = helper.metta.borrow_mut();
149            let prompt = metta.get_config_string(CFG_PROMPT).expect("Fatal Error: Invalid REPL config");
150            let styled_prompt = metta.get_config_string(CFG_STYLED_PROMPT).unwrap_or_else(|| format!("\x1b[1;32m{prompt}\x1b[0m")); //TODO, Let this default be set inside builtin_init_metta_code() when strings can be parsed with escape chars
151            helper.colored_prompt = styled_prompt;
152            prompt
153        };
154
155        let readline = rl.readline(&prompt);
156        match readline {
157            Ok(line) => {
158                rl.add_history_entry(line.as_str())?;
159
160                let mut metta = rl.helper().unwrap().metta.borrow_mut();
161                metta.exec(line.as_str());
162                metta.print_result();
163            }
164            Err(ReadlineError::Interrupted) |
165            Err(ReadlineError::Eof) => {
166                break;
167            }
168            Err(err) => {
169                println!("Error: {err:?}");
170                break;
171            }
172        }
173    }
174
175    if let Some(history_path) = &repl_params.history_file {
176        rl.append_history(history_path)?
177    }
178
179    Ok(())
180}

182struct EnterKeyHandler {
183    force_submit: Arc<Mutex<bool>>
184}
185
186impl EnterKeyHandler {
187    fn new(force_submit: Arc<Mutex<bool>>) -> Self {
188        Self {
189            force_submit
190        }
191    }
192}
193
194impl ConditionalEventHandler for EnterKeyHandler {
195    fn handle(
196        &self,
197        _evt: &Event,
198        _n: RepeatCount,
199        _positive: bool,
200        _ctx: &EventContext<'_>
201    ) -> Option<Cmd> {
202        *self.force_submit.lock().unwrap() = true;
203        Some(Cmd::AcceptOrInsertLine {
204            accept_in_the_middle: true,
205        })
206    }
