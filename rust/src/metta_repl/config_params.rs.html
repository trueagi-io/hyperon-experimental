10pub const CFG_PROMPT: &str = "&ReplPrompt";
11pub const CFG_STYLED_PROMPT: &str = "&ReplStyledPrompt";
12pub const CFG_BRACKET_STYLES: &str = "&ReplBracketStyles";
13pub const CFG_COMMENT_STYLE: &str = "&ReplCommentStyle";
14pub const CFG_VARIABLE_STYLE: &str = "&ReplVariableStyle";

18pub const CFG_BRACKET_MATCH_STYLE: &str = "&ReplBracketMatchStyle";
19pub const CFG_BRACKET_MATCH_ENABLED: &str = "&ReplBracketMatchEnabled";
20pub const CFG_HISTORY_MAX_LEN: &str = "&ReplHistoryMaxLen";
21
22#[derive(Default, Debug)]
23pub struct ReplParams {
24    /// A path to the repl.metta file that's run to configure the repl environment
25    pub repl_config_metta_path: Option<PathBuf>,
26
27    /// A file for previous statements in the interactive repl
28    pub history_file: Option<PathBuf>,
29}
30
31impl ReplParams {
32    pub fn new(shim: &MettaShim) -> Self {
33
34        if let Some(config_dir) = shim.config_dir() {
35
36            //Create the default repl.meta file, if it doesn't already exist
37            let repl_config_metta_path = config_dir.join("repl.metta");
38            if !repl_config_metta_path.exists() {
39                let mut file = fs::OpenOptions::new()
40                    .create(true)
41                    .write(true)
42                    .open(&repl_config_metta_path)
43                    .expect(&format!("Error creating default repl config file at {repl_config_metta_path:?}"));
44                file.write_all(&DEFAULT_REPL_METTA).unwrap();
45            }
46
47            Self {
48                repl_config_metta_path: Some(repl_config_metta_path),
49                history_file: Some(config_dir.join("history.txt")),
50            }
51        } else {
52            Self {
53                repl_config_metta_path: None,
54                history_file: None,
55            }
56        }
57    }
58
59}
