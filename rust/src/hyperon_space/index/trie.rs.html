13/// Implementer of the duplication strategy.
14pub trait DuplicationStrategyImplementor {
15    /// Return reference to the items counter.
16    fn dup_counter_mut(&mut self) -> &mut usize;
17}

19/// Duplication strategy type.
20// TODO: modify duplication strategy to be able represent TrieKey::Leaf differently
21pub trait DuplicationStrategy: Default {
22    fn add_atom(leaf: &mut dyn DuplicationStrategyImplementor);
23    fn remove_atom(leaf: &mut dyn DuplicationStrategyImplementor);
24}

26/// Duplication strategy which forbids duplication.
27#[derive(Debug, Default, PartialEq, Clone)]
28pub struct NoDuplication {}
29impl DuplicationStrategy for NoDuplication {
30    fn add_atom(leaf: &mut dyn DuplicationStrategyImplementor) {
31        let count = leaf.dup_counter_mut();
32        *count = 1;
33    }
34    fn remove_atom(leaf: &mut dyn DuplicationStrategyImplementor) {
35        let count = leaf.dup_counter_mut();
36        *count = 0;
37    }
38}
39
40impl Display for NoDuplication {
41    fn fmt(&self, f: &mut Formatter) -> std::fmt::Result {
42        write!(f, "NoDuplication")
43    }
44}

46/// Duplication strategy which allows duplication.
47#[derive(Debug, Default, PartialEq, Clone)]
48pub struct AllowDuplication {}
49impl DuplicationStrategy for AllowDuplication {
50    fn add_atom(leaf: &mut dyn DuplicationStrategyImplementor) {
51        let count = leaf.dup_counter_mut();
52        *count += 1;
53    }
54    fn remove_atom(leaf: &mut dyn DuplicationStrategyImplementor) {
55        let count = leaf.dup_counter_mut();
56        *count -= 1;
57    }
58}

60impl Display for AllowDuplication {
61    fn fmt(&self, f: &mut Formatter) -> std::fmt::Result {
62        write!(f, "AllowDuplication")
63    }
64}

81/// Key for atoms querying.
82#[derive(Debug)]
83pub enum QueryKey<'a> {
84    /// Start expression token with expression to be queried.
85    StartExpr(&'a Atom, usize),
86    /// Borrowed atom token.
87    Atom(&'a Atom),
88}
89
90impl QueryKey<'_> {
91    /// Consumes key until the end of the current expression. It assumes start
92    /// token of the expression is already consumed.
93    fn skip_expression<'a, I: Iterator<Item=QueryKey<'a>>>(key: &mut I, mut count: usize) {
94        while count > 0 {
95            match key.next() {
96                Some(QueryKey::StartExpr(_, size)) => count = count + size - 1,
97                Some(_) => count -= 1,
98                None => break,
99            }
100        }
101    }

105/// Storage for the atoms from the [AtomTrie] which maps atoms to the [TrieKey] and
106/// back. It consists of two storages. One for storing hashable atoms which
107/// allows atoms extraction. Second for storing non-hashable atoms which doesn't
108/// allow atoms extraction. It also manually maps two special keys: expression
109/// start and expression end.
110#[derive(Default, Debug, Clone, PartialEq, Eq)]
111struct TrieKeyStorage {
112    hashable: HashableStorage,
113    vec: HoleyVec<Atom>,
114}
115
116impl TrieKeyStorage {
117    /// Add key into the storage and return corresponding [TrieKey].
118    pub fn insert_key(&mut self, key: InsertKey) -> TrieKey {
119        match key {
120            InsertKey::StartExpr(size) => TrieKey::start_expr(size),
121            InsertKey::Atom(atom @ Atom::Variable(_)) => self.add_atom(AtomMatchMode::Unification, atom),
122            InsertKey::Atom(atom @ Atom::Grounded(_)) => {
123                match &atom {
124                    Atom::Grounded(gnd) =>
125                        if gnd.as_grounded().as_match().is_some() {
126                            self.add_atom(AtomMatchMode::Unification, atom)
127                        } else {
128                            self.add_atom(AtomMatchMode::Equality, atom)
129                        }
130                    _ => unreachable!(),
131                }
132            }
133            InsertKey::Atom(atom) => self.add_atom(AtomMatchMode::Equality, atom),
134        }
135    }
136
137    fn add_atom(&mut self, match_mode: AtomMatchMode, atom: Atom) -> TrieKey {
138        self.hashable.insert(atom).map_or_else(|atom| {
139            // There is a trade off between keeping non-hashable but equality
140            // matchable keys (for instance grounded atom without both matching
141            // and serialization) in a collection for equality matching or in a
142            // collection for a unification matching. When they are kept in equality
143            // matching collection then each occurrence of such key in query
144            // will go through all equality entries + all unifiable entries. If they
145            // are kept in a unification matching collection (see TrieNode)
146            // then when equality key is matched it is matched via all such keys
147            // as well while no unification matching with them is possible. Current
148            // code keeps such keys in equality matching collection which slows
149            // down their matching. User should be advised to implement
150            // serialization for the grounded types which don't have unification
151            // matching to make them hashable.
152            TrieKey::new(TrieKeyStore::Index, match_mode, self.vec.push(atom))
153        }, |hash_id| {
154            TrieKey::new(TrieKeyStore::Hash, match_mode, hash_id)
155        })
156    }
157    
158    /// Check if key is in the storage and return matching mode, [TrieKey] if it is
159    /// possible and reference to the atom.
160    pub fn query_key<'a>(&self, key: &QueryKey<'a>) -> (AtomMatchMode, Option<TrieKey>, Option<&'a Atom>) {
161        match key {
162            QueryKey::StartExpr(atom, size) => (AtomMatchMode::Equality, Some(TrieKey::start_expr(*size)), Some(atom)),
163            QueryKey::Atom(atom @ Atom::Variable(_)) => self.get_key(AtomMatchMode::Unification, atom),
164            QueryKey::Atom(atom @ Atom::Grounded(gnd)) if gnd.as_grounded().as_match().is_some() =>
165                self.get_key(AtomMatchMode::Unification, atom),
166            QueryKey::Atom(atom) => self.get_key(AtomMatchMode::Equality, atom),
167        }
168    }
169
170    fn get_key<'a>(&self, match_mode: AtomMatchMode, atom: &'a Atom) -> (AtomMatchMode, Option<TrieKey>, Option<&'a Atom>) {
171        let key = self.hashable.get_id(atom)
172            .map(|hash_id| TrieKey::new(TrieKeyStore::Hash, match_mode, hash_id));
173        (match_mode, key, Some(atom))
174    }
175
176    /// Returns atom from storage by [TrieKey].
177    pub unsafe fn get_atom_unchecked(&self, key: TrieKey) -> &Atom {
178        match key.store() {
179            TrieKeyStore::Hash => self.hashable.get_atom(key.value()).unwrap(),
180            TrieKeyStore::Index => self.vec.get_unchecked(key.value()),
181        }
182    }
183
184    /// Remove [TrieKey] from the store
185    pub fn remove_key(&mut self, key: TrieKey) {
186        // TODO: Ability to remove hashable keys is to be added. To implement
187        // removing hashable keys we need to track how many nodes contains this
188        // key. When key is not hashable it is not reused and each instance is
189        // kept in exacly one node.
190        if key.store() == TrieKeyStore::Index {
191            self.vec.remove(key.value());
192        }
193    }
194}

230    /// Insert list of [InsertKey] into the trie.
231    #[inline]
232    pub fn insert<I: Iterator<Item=InsertKey>>(&mut self, key: I) {
233        self.insert_internal(self.root, key)
234    }

236    fn insert_internal<I: Iterator<Item=InsertKey>>(&mut self, node_id: NodeId, mut key: I) {
237        match key.next() {
238            Some(head) => {
239                let head = self.keys.insert_key(head);
240                match self.index.get(&(node_id, head)) {
241                    Some(child_id) => self.insert_internal(*child_id, key),
242                    None => {
243                        let child_id = self.new_branch(key);
244                        self.nodes[node_id].push(head);
245                        self.index.insert((node_id, head), child_id);
246                    },
247                }
248            },
249            None => D::add_atom(&mut self.nodes[node_id]),
250        }
251    }
252
253    fn new_branch<'a, I: Iterator<Item=InsertKey>>(&mut self, key: I) -> NodeId {
254        let child_id = self.nodes.push(Default::default());
255        self.insert_internal(child_id, key);
256        child_id
257    }

259    /// Query trie using list of the [QueryKey] instances.
260    #[inline]
261    pub fn query<'a, I: Debug + Clone + Iterator<Item=QueryKey<'a>>>(&self, key: I) -> BindingsSet {
262        let mut mapper = CachingMapper::new(|v: &VariableAtom| v.clone().make_unique());
263        self.query_internal(self.root, key, &mut mapper)
264    }
265
266    // TODO: write an algorithm which returns an iterator instead of collected result
267    fn query_internal<'a, I, M>(&self, node_id: NodeId, mut key: I,
268        mapper: &mut CachingMapper<VariableAtom, VariableAtom, M>) -> BindingsSet
269        where
270            I: Debug + Clone + Iterator<Item=QueryKey<'a>>,
271            M: Fn(&VariableAtom)->VariableAtom
272    {
273        match key.next() {
274            Some(head) => {
275                match self.keys.query_key(&head) {
276                    (AtomMatchMode::Equality, head, atom) =>
277                        self.match_key_by_equality(node_id, atom, head, key, mapper),
278                    (AtomMatchMode::Unification, _head, Some(atom)) =>
279                        self.match_key_by_unification(node_id, atom, key, mapper),
280                    (AtomMatchMode::Unification, _head, None) => unreachable!(),
281                }
282            },
283            None => {
284                BindingsSet::count(self.nodes[node_id].leaf_counter())
285            }
286        }
287    }
288
289    fn match_key_by_equality<'a, I, M>(&self, node_id: NodeId,
290        atom: Option<&'a Atom>, key: Option<TrieKey>, mut tail: I,
291        mapper: &mut CachingMapper<VariableAtom, VariableAtom, M>) -> BindingsSet
292        where
293            I: Debug + Clone + Iterator<Item=QueryKey<'a>>,
294            M: Fn(&VariableAtom)->VariableAtom,
295    {
296        let mut result = BindingsSet::empty();
297        if let TrieNode::Leaf(_count) = self.nodes[node_id] {
298            return result;
299        }
300        
301        if let Some(equality_key) = key {
302            if let Some(&child_id) = self.index.get(&(node_id, equality_key)) {
303                result.extend(self.query_internal(child_id, tail.clone(), mapper))
304            }
305        } else {
306            // match equality nonhashable key (see TrieKeyStorage::add_atom)
307            if let Some(query) = atom {
308                let it = self.nodes[node_id].iter_match(AtomMatchMode::Equality)
309                    // FIXME: should we check !is_hashable here?
310                    .filter(|&(_index, key)| !key.is_start_expr())
311                    .map(|(_index, key)| (unsafe{ self.keys.get_atom_unchecked(key) }, key));
312                for (entry, key) in it {
313                    if entry == query {
314                        let child_id = *self.index.get(&(node_id, key)).unwrap();
315                        result.extend(self.query_internal(child_id, tail.clone(), mapper));
316                    }
317                }
318            }
319        }
320
321        // match unification matching entries with equality key
322        if let Some(query) = atom {
323            if let Some(expr_size) = key.and_then(|k| k.as_start_expr()) {
324                QueryKey::skip_expression(&mut tail, expr_size);
325            }
326            let it = self.nodes[node_id].iter_match(AtomMatchMode::Unification)
327                .map(|(_index, key)| {
328                    let entry = unsafe{ self.keys.get_atom_unchecked(key) };
329                    let child_id = *self.index.get(&(node_id, key)).unwrap();
330                    (entry, child_id)
331                });
332            for (entry, child_id) in it {
333                let mut unify_res = self.unify_entry(entry, query, child_id, tail.clone(), mapper);
334                result.extend(unify_res.drain(..));
335            }
336        }
337        result
338    }
339
340    fn unify_entry<'a, I, M>(&self, entry: &Atom, key: &Atom, child_id: NodeId, tail: I,
341        mapper: &mut CachingMapper<VariableAtom, VariableAtom, M>) -> BindingsSet
342        where
343            I: Debug + Clone + Iterator<Item=QueryKey<'a>>,
344            M: Fn(&VariableAtom)->VariableAtom
345    {
346        let mut result = BindingsSet::empty();
347        let mut entry = entry.clone();
348        // TODO: replacing variables each time could be eliminated
349        entry.iter_mut().filter_type::<&mut VariableAtom>().for_each(|var| *var = mapper.replace(var));
350        // TODO: conversion to Iterator and back
351        result.extend(match_atoms(&entry, key));
352        if result.is_empty() {
353            result
354        } else {
355            let tail_result = self.query_internal(child_id, tail, mapper);
356            // TODO: we could move BindingsSet into merge instead of passing by reference
357            result.merge(&tail_result).into_iter().filter(|b| !b.has_loops()).collect()
358        }
359    }
360
361    fn match_key_by_unification<'a, I, M>(&self, node_id: NodeId,
362        atom: &Atom, tail: I,
363        mapper: &mut CachingMapper<VariableAtom, VariableAtom, M>) -> BindingsSet
364        where
365            I: Debug + Clone + Iterator<Item=QueryKey<'a>>,
366            M: Fn(&VariableAtom)->VariableAtom
367    {
368        let mut result = BindingsSet::empty();
369        for (entry, child_id) in self.unpack_atoms_internal(node_id) {
370            let mut tail_result = self.unify_entry(&entry, atom, child_id, tail.clone(), mapper);
371            result.extend(tail_result.drain(..));
372        }
373        result
374    }

399    pub fn remove_internal<'a, I>(&mut self, node_id: NodeId, mut key: I) -> bool
400        where I: Iterator<Item=QueryKey<'a>>
401    {
402        match key.next() {
403            Some(head) => {
404                let entry = match self.keys.query_key(&head) {
405                    (_match, None, None) => None,
406                    (match_mode, Some(key), _atom) => {
407                        match self.index.get(&(node_id, key)) {
408                            Some(child_id) => {
409                                self.nodes[node_id].iter_match(match_mode)
410                                    .find(|(_i, k)| *k == key)
411                                    .map(|(i, k)| (k, i, *child_id))
412                            },
413                            None => None,
414                        }
415                    },
416                    (match_mode, None, Some(atom)) => {
417                        self.nodes[node_id].iter_match(match_mode)
418                            .find(|(_i, k)| atom == unsafe{ self.keys.get_atom_unchecked(*k) })
419                            .map(|(i, k)| (k, i, *self.index.get(&(node_id, k)).unwrap()))
420                    },
421                };
422                match entry {
423                    Some((child_key, index, child_id)) => {
424                        let removed = self.remove_internal(child_id, key);
425                        if removed && self.nodes[child_id].is_leaf() {
426                            self.index.remove(&(node_id, child_key));
427                            self.nodes[node_id].remove_key(child_key, index);
428                            self.keys.remove_key(child_key);
429                        }
430                        removed
431                    },
432                    None => false,
433                }
434            },
435            None => {
436                D::remove_atom(&mut self.nodes[node_id]);
437                true
438            },
439        }
440    }

455impl<D: DuplicationStrategy> AtomTrie<D> {
456    fn fmt_node(&self, level: usize, node_id: NodeId, f: &mut Formatter) -> std::fmt::Result {
457        let node = &self.nodes[node_id];
458        if let TrieNode::Leaf(count) = node {
459            write!(f, "{}{}\n", " ".repeat(level), count)
460        } else {
461            node.iter_all().fold(Ok(()), |rc, (_, key)| {
462                let child_id = self.index.get(&(node_id, key)).unwrap();
463                rc.and_then(|_| self.fmt_key(level, *child_id, key, f))
464                    .and_then(|_| {
465                        self.fmt_node(level + 1, *child_id, f)
466                    })
467            })
468        }
469    }

471    fn fmt_key(&self, level: usize, node_id: NodeId, key: TrieKey, f: &mut Formatter) -> std::fmt::Result {
472        match key.as_start_expr() {
473            Some(size) => write!(f, "{}({} [{}] <{:?}>\n", " ".repeat(level), size, node_id, key),
474            None => {
475                let atom = unsafe{ self.keys.get_atom_unchecked(key) };
476                write!(f, "{}{} [{}] <{:?}>\n", " ".repeat(level), atom, node_id, key)
477            },
478        }
479    }

482/// Which storage keeps the value of the key.
483#[derive(Debug, Clone, Copy, PartialEq, Eq)]
484enum TrieKeyStore {
485    /// Value is kept in the storage for hashable atoms.
486    Hash,
487    /// Value is kept in the storage for nonhashable atoms.
488    Index,
489}
490
491/// How atom represented by the key should be matched.
492#[derive(Debug, Clone, Copy, PartialEq, Eq)]
493enum AtomMatchMode {
494    /// Atom is matched using equality.
495    Equality,
496    /// Atom is matched using unification.
497    Unification,
498}
499
500const BITS_PER_ID: u32 = usize::BITS - 2;
501
502const TK_STORE_MASK: usize = 0b10 << BITS_PER_ID;
503const TK_MATCH_MASK: usize = 0b01 << BITS_PER_ID;
504const TK_VALUE_MASK: usize = !(0b11 << BITS_PER_ID);
505
506const TK_STORE_HASH: usize = 0b00 << BITS_PER_ID;
507const TK_STORE_INDEX: usize = 0b10 << BITS_PER_ID;
508const TK_MATCH_EXACT: usize = 0b00 << BITS_PER_ID;
509const TK_MATCH_CUSTOM: usize = 0b01 << BITS_PER_ID;
510
511const TK_MAX_EXPRESSION_SIZE: usize = 1 << 10;
512
513/// Compact representation of the atom from the trie. It represents each
514/// atom using single [usize] value. It keeps value of the key, key matching
515/// mode: equality or unification, key storage mode: hashable or indexed.
516#[derive(Hash, Copy, Clone, PartialEq, Eq)]
517struct TrieKey(usize);
518
519impl TrieKey {
520    #[allow(non_snake_case)]
521    #[inline]
522    const fn new(store: TrieKeyStore, match_mode: AtomMatchMode, mut value: usize) -> Self {
523        let store = match store {
524            TrieKeyStore::Hash => TK_STORE_HASH,
525            TrieKeyStore::Index => TK_STORE_INDEX,
526        };
527        let match_mode = match match_mode {
528            AtomMatchMode::Equality => TK_MATCH_EXACT,
529            AtomMatchMode::Unification => TK_MATCH_CUSTOM,
530        };
531        value += TK_MAX_EXPRESSION_SIZE;
532        assert!(((!TK_VALUE_MASK) & value) == 0);
533        Self(store | match_mode | value)
534    }
535
536    #[allow(non_snake_case)]
537    #[inline]
538    const fn start_expr(size: usize) -> Self {
539        assert!(size < TK_MAX_EXPRESSION_SIZE);
540        Self(TK_STORE_HASH | TK_MATCH_EXACT | size)
541    }
542
543    #[inline]
544    fn value(&self) -> usize {
545        self.0 & TK_VALUE_MASK - TK_MAX_EXPRESSION_SIZE
546    }
547
548    #[inline]
549    fn store(&self) -> TrieKeyStore {
550        match self.0 & TK_STORE_MASK {
551            TK_STORE_HASH => TrieKeyStore::Hash,
552            TK_STORE_INDEX => TrieKeyStore::Index,
553            _ => unreachable!(),
554        }
555    }
556
557    #[inline]
558    fn match_mode(&self) -> AtomMatchMode {
559        match self.0 & TK_MATCH_MASK {
560            TK_MATCH_EXACT => AtomMatchMode::Equality,
561            TK_MATCH_CUSTOM => AtomMatchMode::Unification,
562            _ => unreachable!(),
563        }
564    }
565
566    #[inline]
567    fn is_start_expr(&self) -> bool {
568        self.0 < TK_MAX_EXPRESSION_SIZE
569            && self.store() == TrieKeyStore::Hash
570            && self.match_mode() == AtomMatchMode::Equality
571    }
572
573    #[inline]
574    fn as_start_expr(&self) -> Option<usize> {
575        if self.is_start_expr() {
576            Some(self.0 & TK_VALUE_MASK)
577        } else {
578            None
579        }
580    }
581}

583impl Debug for TrieKey {
584    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
585        if self.is_start_expr() {
586            write!(f, "TrieKey{{ expr: {:?} }}", self.as_start_expr().unwrap())
587        } else {
588            write!(f, "TrieKey{{ match_mode: {:?}, store: {:?}, value: {:?} }}",
589                self.match_mode(), self.store(), self.value())
590        }
591    }

594/// [AtomTrie] node representation. Node doesn't keep a map between keys and
595/// successors. It keeps just a list of keys. The map is kept in the [AtomTrie]
596/// itself. It is much more compact memory representation. List of keys in the
597/// node is required to iterate over successors.
598#[derive(Debug, Clone, PartialEq, Eq)]
599enum TrieNode {
600    /// Leaf node which contains the counter of atoms.
601    Leaf(usize),
602    /// Single node which contains the only successor.
603    Single(TrieKey),
604    /// Collection of the successors.
605    Many(Box<TrieNodeKeys>),
606}
607
608/// [TrieNode] successors collection which keeps separately keys matched
609/// by equality and keys matched by unification.
610#[derive(Default, Debug, Clone, PartialEq, Eq)]
611struct TrieNodeKeys {
612    equal: smallvec::SmallVec<[TrieKey; 2]>,
613    unify: smallvec::SmallVec<[TrieKey; 0]>,
614}
615
616impl TrieNodeKeys {
617    fn push(&mut self, key: TrieKey) {
618        match key.match_mode() {
619            AtomMatchMode::Equality => self.equal.push(key),
620            AtomMatchMode::Unification => self.unify.push(key),
621        }
622    }
623
624    fn remove(&mut self, key: TrieKey, index: usize) {
625        match key.match_mode() {
626            AtomMatchMode::Equality => self.equal.remove(index),
627            AtomMatchMode::Unification => self.unify.remove(index),
628        };
629    }
630
631    fn len(&self) -> usize {
632        self.equal.len() + self.unify.len()
633    }
634
635    fn single(&self) -> TrieKey {
636        if self.equal.len() == 1 {
637            self.equal[0]
638        } else if self.unify.len() == 1 {
639            self.unify[0]
640        } else {
641            panic!("Collection contains more than 1 item");
642        }
643    }
644
645    fn iter_match(&self, match_mode: AtomMatchMode) -> std::slice::Iter<'_, TrieKey> {
646        match match_mode {
647            AtomMatchMode::Equality => self.equal.iter(),
648            AtomMatchMode::Unification => self.unify.iter(),
649        }
650    }
651}
652
653impl DuplicationStrategyImplementor for TrieNode {
654    fn dup_counter_mut(&mut self) -> &mut usize {
655        self.leaf_counter_mut()
656    }
657}
658
659impl Default for TrieNode {
660    fn default() -> Self {
661        TrieNode::Leaf(0)
662    }
663}
664
665impl TrieNode {
666    /// Return items counter from leaf node.
667    pub fn leaf_counter(&self) -> usize {
668        match self {
669            TrieNode::Leaf(count) => *count,
670            // This invariant holds only for keys which are properly balanced
671            // i.e. for keys constructed from atoms
672            _ => panic!("Key is expected to end by leaf node"),
673        }
674    }
675
676    /// Return mutable items counter from leaf node.
677    fn leaf_counter_mut(&mut self) -> &mut usize {
678        match self {
679            TrieNode::Leaf(count) => count,
680            // This invariant holds only for keys which are properly balanced
681            // i.e. for keys constructed from atoms
682            _ => panic!("Key is expected to end by leaf node"),
683        }
684    }
685
686    /// Return `true` if node is a leaf.
687    pub fn is_leaf(&self) -> bool {
688        matches!(*self, TrieNode::Leaf(0))
689    }
690
691    /// Push new key into a list of successors.
692    pub fn push(&mut self, key: TrieKey) {
693        match self {
694            Self::Leaf(0) => *self = Self::Single(key),
695            Self::Leaf(_) => panic!("Unexpected state"),
696            Self::Single(prev) => {
697                let mut keys: TrieNodeKeys = Default::default();
698                keys.push(*prev);
699                keys.push(key);
700                *self = Self::Many(Box::new(keys))
701            },
702            Self::Many(keys) => keys.push(key),
703        }
704
705
706    }
707
708    /// Remove key from the node.
709    pub fn remove_key(&mut self, key: TrieKey, idx: usize) {
710        match self {
711            Self::Leaf(_) => panic!("Unexpected state"),
712            Self::Single(k) => {
713                assert!(idx == 0, "Unexpected state");
714                assert!(*k == key);
715                *self = Self::default();
716            },
717            Self::Many(keys) => {
718                keys.remove(key, idx);
719                if keys.len() == 1 {
720                    *self = Self::Single(keys.single());
721                }
722            }
723        }
724    }
725
726    /// Iterate over all of keys in the list
727    pub fn iter_all(&self) -> std::iter::Chain<TrieNodeIndexIter<'_>, TrieNodeIndexIter<'_>> {
728        self.iter_match(AtomMatchMode::Unification)
729            .chain(self.iter_match(AtomMatchMode::Equality))
730    }
731
732    /// Iterate over keys which has specified matching mode.
733    pub fn iter_match(&self, match_mode: AtomMatchMode) -> TrieNodeIndexIter<'_> {
734        match self {
735            Self::Leaf(_) => TrieNodeIndexIter::Empty,
736            Self::Single(key) => {
737                if key.match_mode() == match_mode {
738                    TrieNodeIndexIter::Single(*key)
739                } else {
740                    TrieNodeIndexIter::Empty
741                }
742            },
743            Self::Many(keys) => TrieNodeIndexIter::Many(keys
744                    .iter_match(match_mode)
745                    .copied()
746                    .enumerate())
747        }
748    }
749}
750
751/// Iterator over [TrieNode] keys and corresponding indexes.
752#[derive(Debug, Clone)]
753enum TrieNodeIndexIter<'a> {
754    Empty,
755    Single(TrieKey),
756    Many(std::iter::Enumerate<std::iter::Copied<std::slice::Iter<'a, TrieKey>>>),
757}
758
759impl Iterator for TrieNodeIndexIter<'_> {
760    type Item = (usize, TrieKey);
761
762    fn next(&mut self) -> Option<Self::Item> {
763        match self {
764            Self::Empty => None,
765            Self::Single(key) => {
766                let key = *key;
767                *self = Self::Empty;
768                Some((0, key))
769            },
770            Self::Many(it) => {
771                it.next()
772            }
773        }
774    }
775}
776
777type TrieNodeIter<'a> = std::iter::Chain<TrieNodeIndexIter<'a>, TrieNodeIndexIter<'a>>;

779#[derive(Debug)]
780struct ExpressionBuilder {
781    count: usize,
782    children: Vec<Atom>,
783}
784
785struct TrieNodeAtomIter<'a, D: DuplicationStrategy> {
786    trie: &'a AtomTrie<D>,
787    node_id: NodeId,
788    iter: TrieNodeIter<'a>,
789    path: Vec<(NodeId, TrieNodeIter<'a>)>,
790    expr: Vec<ExpressionBuilder>,
791}
792
793impl<'a, D: DuplicationStrategy> TrieNodeAtomIter<'a, D> {
794    fn new(trie: &'a AtomTrie<D>, node_id: NodeId) -> Self {
795        Self {
796            trie,
797            node_id,
798            iter: trie.nodes[node_id].iter_all(),
799            path: Vec::new(),
800            expr: Vec::new(),
801        }
802    }
803
804    fn push(&mut self, subnode_id: NodeId, key: TrieKey)  {
805        match key.as_start_expr() {
806            None => {
807                let atom = unsafe{ self.trie.keys.get_atom_unchecked(key) };
808                let expr = self.expr.last_mut().unwrap();
809                expr.children.push(atom.clone());
810                expr.count -= 1;
811            },
812            Some(expr_size) => {
813                self.expr.push(ExpressionBuilder {
814                    count: expr_size,
815                    children: Vec::new(),
816                });
817            },
818        }
819
820        self.collapse_expressions();
821
822        let prev = std::mem::replace(&mut self.iter, self.trie.nodes[subnode_id].iter_all());
823        self.path.push((self.node_id, prev));
824        self.node_id = subnode_id;
825    }
826
827    fn collapse_expressions(&mut self) {
828        loop {
829            if self.expr.len() == 1
830                || self.expr.last_mut().unwrap().count > 0 {
831                    break
832            }
833            let expr = self.expr.pop().unwrap();
834            let last = self.expr.last_mut().unwrap();
835            last.children.push(Atom::expr(expr.children));
836            last.count -= 1;
837        }
838    }
839
840    fn pop(&mut self) -> bool {
841        match self.path.pop() {
842            Some(prev) => {
843                self.expand_expressions();
844
845                let expr = self.expr.last_mut().unwrap();
846                match expr.children.pop() {
847                    Some(_) => {
848                        expr.count += 1;
849                    },
850                    None => {
851                        self.expr.pop();
852                    },
853                }
854
855                (self.node_id, self.iter) = prev;
856                true
857            },
858            None => false,
859        }
860    }
861
862    fn expand_expressions(&mut self) {
863        loop {
864            let last = self.expr.last_mut().unwrap();
865            let children = &mut last.children;
866            if let Some(Atom::Expression(_)) = children.last() {
867                last.count += 1;
868                let atom = children.pop().unwrap();
869                let vec = ExpressionAtom::try_from(atom).unwrap().into_children();
870                self.expr.push(ExpressionBuilder {
871                    count: 0,
872                    children: vec
873                });
874            } else {
875                break
876            }
877        }
878    }
879
880}
881
882impl<'a, D: DuplicationStrategy> Iterator for TrieNodeAtomIter<'a, D> {
883    type Item = (Cow<'a, Atom>, NodeId);
884
885    fn next(&mut self) -> Option<Self::Item> {
886        loop {
887            if self.expr.len() == 1 {
888                let expr = self.expr.last_mut().unwrap();
889                if expr.count == 0 {
890                    let atom = Atom::expr(expr.children.clone());
891                    let node_id = self.node_id;
892                    self.pop();
893                    return Some((Cow::Owned(atom), node_id))
894                }
895            }
896
897            match self.iter.next() {
898                None => {
899                    if !self.pop() {
900                        return None
901                    }
902                },
903                Some((_, key)) => {
904                    let sub_node_id = *self.trie.index.get(&(self.node_id, key)).unwrap();
905                    if self.expr.is_empty() && !key.is_start_expr() {
906                        let atom = unsafe{ self.trie.keys.get_atom_unchecked(key) };
907                        return Some((Cow::Borrowed(atom), sub_node_id))
908                    } else {
909                        self.push(sub_node_id, key);
910                    }
911                },
912            }
913        }
914    }
915}
